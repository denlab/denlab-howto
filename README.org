** ssh or scp in a script without entering password ?

#+begin_src sh
expect -c "
   spawn scp login@somehost.com:/somefile .
   expect yes/no { send yes\r ; exp_continue }
   expect password: { send password\r }
   expect 100%
   exit
"
#+end_src
